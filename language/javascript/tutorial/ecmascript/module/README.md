# 循环依赖
- CommonJS：可以在引入其他模块前输出

    - 命名输出

        ```
        // --- a.js ---
        var b = require('b');
        exports.foo = function () { ... };
        // --- b.js ---
        var a = require('a'); // (1)
        // Can’t use a.foo in module body,
        // but it will be filled in later
        exports.bar = function () {
          a.foo(); // OK (2)
        };
        // --- main.js ---
        var a = require('a');
        ```

    - 默认输出

        ```
        // --- a.js ---
        var b = require('b');
        exports.exports = function () { ... };
        // --- b.js ---
        var a = require('a'); // (1)
        exports.bar = function () {
          a.foo(); // Fail (2)
        };
        // --- main.js ---
        var a = require('a');
        ```
